# Feature Card

Card component for highlighting features.

## Usage

```tsx
import { Card, CardTitle, CardContent } from "@/components/ui/card"

export default function Example() {
  return (
    <Card>
      <CardContent>
        <CardTitle>Feature</CardTitle>
        <p>Feature description goes here.</p>
      </CardContent>
    </Card>
  )
}
```
